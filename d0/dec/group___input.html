User defines what happens when a specific key is pressed.

Example:
void Ship::OnKeyPress(AZUL_KEY k)
{
switch (k)
{
case(AZUL_KEY::KEY_T):
DebugMsg::out("T key pressed!\n");
BulletFactory::CreateBullet(this);
break;
default:
break;
}
}
